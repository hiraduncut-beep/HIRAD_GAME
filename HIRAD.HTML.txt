function initAudio() {
  if (!audioCtx) {

// ========== صدای تیر ==========
function playShootSound(volume = 0.15) {
  if (!soundEnabled || !audioCtx) return;
  try {
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = 'square';
    osc.frequency.setValueAtTime(800, audioCtx.currentTime);
    osc.frequency.exponentialRampToValueAtTime(200, audioCtx.currentTime + 0.05);
    gain.gain.setValueAtTime(volume, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.05);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start();
    osc.stop(audioCtx.currentTime + 0.05);
    
    // نویز اضافه برای صدای واقعیتر
    const bufferSize = audioCtx.sampleRate * 0.03;
    const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
    const data = buffer.getChannelData(0);
    for (let i = 0; i < bufferSize; i++) {
      data[i] = (Math.random() * 2 - 1) * Math.exp(-i / bufferSize * 3);
    }
    const noise = audioCtx.createBufferSource();
    noise.buffer = buffer;
    const noiseGain = audioCtx.createGain();
    noiseGain.gain.setValueAtTime(volume * 0.3, audioCtx.currentTime);
    noiseGain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.03);
    noise.connect(noiseGain);
    noiseGain.connect(audioCtx.destination);
    noise.start();
    noise.stop(audioCtx.currentTime + 0.03);
  } catch(e) {}
}

// ========== صدای انفجار ==========
function playExplosionSound(volume = 0.2) {
  if (!soundEnabled || !audioCtx) return;
  try {
    const bufferSize = audioCtx.sampleRate * 0.2;
    const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
    const data = buffer.getChannelData(0);
    for (let i = 0; i < bufferSize; i++) {
      data[i] = (Math.random() * 2 - 1) * Math.exp(-i / bufferSize * 2);
    }
    const noise = audioCtx.createBufferSource();
    noise.buffer = buffer;
    const gain = audioCtx.createGain();
    gain.gain.setValueAtTime(volume, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.2);
    noise.connect(gain);
    gain.connect(audioCtx.destination);
    noise.start();
    noise.stop(audioCtx.currentTime + 0.2);
  } catch(e) {}
}

// ========== صدای راکت ==========
function playRocketSound() {
  if (!soundEnabled || !audioCtx) return;
  try {
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = 'sawtooth';
    osc.frequency.setValueAtTime(150, audioCtx.currentTime);
    osc.frequency.exponentialRampToValueAtTime(50, audioCtx.currentTime + 0.3);
    gain.gain.setValueAtTime(0.2, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.3);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start();
    osc.stop(audioCtx.currentTime + 0.3);
  } catch(e) {}
}

let musicInterval = null;
let musicNotes = [262, 294, 330, 349, 392, 440, 494, 523];
let musicIndex = 0;

function playMusic() {
  if (!soundEnabled || !audioCtx) return;
  try {
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.type = 'sine';
    const note = musicNotes[musicIndex % musicNotes.length];
    osc.frequency.setValueAtTime(note * (0.5 + Math.sin(gameTime * 0.5) * 0.1), audioCtx.currentTime);
    gain.gain.setValueAtTime(0.03, audioCtx.currentTime);
    gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.15);
    osc.connect(gain);
    gain.connect(audioCtx.destination);
    osc.start();
    osc.stop(audioCtx.currentTime + 0.15);
    musicIndex++;
    if (musicIndex > 20) musicIndex = 0;
  } catch(e) {}
}

function randomRange(min, max) {

}

// ========== تیراندازی ==========
function shootBullet(fromX, fromY, targetX, targetY, isRocket = false) {
  const angle = Math.atan2(targetY - fromY, targetX - fromX);
  const speed = isRocket ? 5 : 8;
  bullets.push({
    x: fromX + Math.cos(angle) * 20,
    y: fromY + Math.sin(angle) * 20,
    vx: Math.cos(angle) * speed,
    vy: Math.sin(angle) * speed,
    radius: isRocket ? 10 : 4,
    damage: isRocket ? 80 : 20,
    life: isRocket ? 80 : 150,
    color: isRocket ? '#ff6633' : '#00ddff',
    trail: [],
    isRocket: isRocket,
    angle: angle
  });
  
  if (isRocket) {
    playRocketSound();
    // ذرات دود راکت
    for (let i = 0; i < 8; i++) {
      particles.push({
        x: fromX + Math.cos(angle) * 20,
        y: fromY + Math.sin(angle) * 20,
        vx: Math.cos(angle + randomRange(-0.5, 0.5)) * randomRange(1, 2),
        vy: Math.sin(angle + randomRange(-0.5, 0.5)) * randomRange(1, 2),
        life: randomRange(5, 15),
        maxLife: 15,
        size: randomRange(2, 5),
        color: `hsl(30, 100%, ${randomRange(50, 80)}%)`,
        gravity: -0.02,
        type: 'spark'
      });
    }
  } else {
    playShootSound(0.1);
  }
}

// ========== انیمیشن اصلی ==========
function animate() {
  gameTime += 0.02;
  
  // ======== پسزمینه ========
  ctx.fillStyle = 'rgba(5, 5, 15, 0.15)';
  ctx.fillRect(0, 0, canvas.width, canvas.height);
  
  // ======== ستارهها ========
  stars.forEach(s => {
    s.twinkle += 0.02;
    const alpha = 0.3 + Math.sin(s.twinkle) * 0.3;
    ctx.beginPath();
    ctx.arc(s.x, s.y, s.size * (0.8 + Math.sin(s.twinkle * 0.5) * 0.2), 0, Math.PI * 2);
    ctx.fillStyle = `rgba(255,255,255,${alpha * s.depth})`;
    ctx.fill();
    s.x += 0.02 * s.depth;
    if (s.x > canvas.width) s.x = 0;
  });
  
  // ======== موجهای ضربه ========
  shockwaves.forEach((sw, idx) => {
    sw.radius += 2.5;
    sw.life -= 0.015;
    if (sw.life <= 0) { shockwaves.splice(idx, 1); return; }
    ctx.beginPath();
    ctx.arc(sw.x, sw.y, sw.radius, 0, Math.PI * 2);
    ctx.strokeStyle = `rgba(255,200,100,${sw.life * 0.25})`;
    ctx.lineWidth = 2 * sw.life;
    ctx.stroke();
  });
  
  // ======== ذرات ========
  particles.forEach((p, idx) => {
    p.x += p.vx;
    p.y += p.vy;
    if (p.gravity) p.vy += p.gravity;
    p.vx *= 0.99;
    p.vy *= 0.99;
    p.life--;
    if (p.life <= 0) { particles.splice(idx, 1); return; }
    const alpha = p.life / p.maxLife;
    ctx.globalAlpha = alpha;
    ctx.shadowColor = p.color;
    ctx.shadowBlur = 15;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.size * (0.3 + alpha * 0.7), 0, Math.PI * 2);
    ctx.fillStyle = p.color;
    ctx.fill();
    ctx.shadowBlur = 0;
    ctx.globalAlpha = 1;
  });
  
  // ======== حرکت قهرمان ========
  const dx = mouseX - hero.x;
  const dy = mouseY - hero.y;
  const dist = Math.sqrt(dx*dx + dy*dy);
  if (dist > 3) {
    const moveSpeed = Math.min(hero.speed, dist);
    hero.x += (dx / dist) * moveSpeed;
    hero.y += (dy / dist) * moveSpeed;
  }
  hero.angle = Math.atan2(dy, dx);
  
  // محدودیت در صفحه
  hero.x = Math.max(hero.radius, Math.min(canvas.width - hero.radius, hero.x));
  hero.y = Math.max(hero.radius, Math.min(canvas.height - hero.radius, hero.y));
  
  // ======== تیراندازی خودکار ========
  if (hero.attackCooldown <= 0 && dist < 450) {
    shootBullet(hero.x, hero.y, mouseX, mouseY, false);
    hero.attackCooldown = 5;
  }
  if (hero.attackCooldown > 0) hero.attackCooldown--;
  
  // ======== بروزرسانی گلولهها ==========
  bullets.forEach((b, idx) => {
    b.x += b.vx;
    b.y += b.vy;
    if (b.isRocket) {
      b.vx *= 1.001;
      b.vy *= 1.001;
    }
    b.life--;
    if (b.life <= 0 || b.x < -50 || b.x > canvas.width + 50 || 
        b.y < -50 || b.y > canvas.height + 50) {
      if (b.isRocket) {
        createExplosion(b.x, b.y, 2.5, '#ff6633');
      }
      bullets.splice(idx, 1);
      return;
    }
    
    // دنباله
    b.trail.push({x: b.x, y: b.y});
    if (b.trail.length > 8) b.trail.shift();
    b.trail.forEach((t, i) => {
      ctx.globalAlpha = i / b.trail.length * 0.3;
      ctx.beginPath();
      ctx.arc(t.x, t.y, b.radius * (i / b.trail.length) * 0.5, 0, Math.PI * 2);
      ctx.fillStyle = b.color;
      ctx.fill();
      ctx.globalAlpha = 1;
    });
    
    // برخورد با دشمنان
    enemies.forEach((e, eIdx) => {
      if (distance(b.x, b.y, e.x, e.y) < b.radius + e.radius) {
        e.health -= b.damage;
        createExplosion(b.x, b.y, b.isRocket ? 2 : 0.4, null);
        if (b.isRocket) {
          // انفجار بزرگتر
          createExplosion(e.x, e.y, 2, '#ff6633');
          // آسیب به دشمنان مجاور
          enemies.forEach((other, oIdx) => {
            if (oIdx !== eIdx && distance(e.x, e.y, other.x, other.y) < 100) {
              other.health -= 30;
            }
          });
        }
        bullets.splice(idx, 1);
        if (e.health <= 0) {
          createExplosion(e.x, e.y, 1.2, e.color);
          spawnItem(e.x, e.y);
          enemies.splice(eIdx, 1);
          score += 15 + (e.type === 'fast' ? 5 : e.type === 'heavy' ? 20 : 10);
          comboCount++;
          if (comboCount > comboDisplay) comboDisplay = comboCount;
          document.getElementById('scoreDisplay').innerHTML = score;
        }
      }
    });
    
    // برخورد با باس
    if (bossActive && boss) {
      if (distance(b.x, b.y, boss.x, boss.y) < b.radius + boss.radius) {
        boss.health -= b.damage;
        createExplosion(b.x, b.y, b.isRocket ? 2 : 0.5, '#ff2266');
        bullets.splice(idx, 1);
        if (boss.health <= 0) {
          createExplosion(boss.x, boss.y, 3, '#ff2266');
          // انفجار بزرگ
          for (let i = 0; i < 3; i++) {
            setTimeout(() => {
              createExplosion(boss.x + randomRange(-80, 80), boss.y + randomRange(-80, 80), 1.5, '#ff2266');
            }, i * 200);
          }
          score += 200;
          document.getElementById('scoreDisplay').innerHTML = score;
          bossActive = false;
          boss = null;
          screenFlash = 0.5;
          shakeAmount = 20;
          playExplosionSound(0.5);
        }
      }
    }
  });
  
  // ======== بروزرسانی دشمنان ==========
  enemies.forEach((e, idx) => {
    const dx2 = hero.x - e.x;
    const dy2 = hero.y - e.y;
    const dist2 = Math.sqrt(dx2*dx2 + dy2*dy2);
    if (dist2 > 0) {
      e.x += (dx2 / dist2) * e.speed;
      e.y += (dy2 / dist2) * e.speed;
    }
    
    // برخورد با قهرمان
    if (dist2 < e.radius + hero.radius) {
      hero.health -= 0.3;
      createExplosion(e.x, e.y, 0.3, '#ff3333');
      if (e.type === 'heavy') {
        e.x -= (dx2 / dist2) * 15;
        e.y -= (dy2 / dist2) * 15;
      }
      if (hero.health <= 0) hero.health = 0;
      document.getElementById('healthDisplay').innerHTML = Math.floor(hero.health);
    }
  });
  
  // ======== بروزرسانی باس ==========
  if (bossActive && boss) {
    boss.phase += 0.02;
    const dx2 = hero.x - boss.x;
    const dy2 = hero.y - boss.y;
    const dist2 = Math.sqrt(dx2*dx2 + dy2*dy2);
    if (dist2 > 0) {
      boss.x += (dx2 / dist2) * boss.speed;
      boss.y += (dy2 / dist2) * boss.speed;
    }
    boss.x += Math.sin(gameTime * 0.5) * 0.3;
    boss.angle = Math.atan2(dy2, dx2);
    
    // برخورد با قهرمان
    if (dist2 < boss.radius + hero.radius) {
      hero.health -= 0.8;
      createExplosion(boss.x, boss.y, 0.5, '#ff2266');
      if (hero.health <= 0) hero.health = 0;
      document.getElementById('healthDisplay').innerHTML = Math.floor(hero.health);
    }
    
    // تیراندازی باس
    boss.shootTimer--;
    if (boss.shootTimer <= 0 && dist2 < 600) {
      const angle = Math.atan2(hero.y - boss.y, hero.x - boss.x);
      // تیرهای باس (۳ تا)
      for (let i = -1; i <= 1; i++) {
        const a = angle + i * 0.2;
        bullets.push({
          x: boss.x + Math.cos(angle) * 30,
          y: boss.y + Math.sin(angle) * 30,
          vx: Math.cos(a) * 3.5,
          vy: Math.sin(a) * 3.5,
          radius: 5,
          damage: 10,
          life: 100,
          color: '#ff0044',
          trail: [],
          isRocket: false
        });
      }
      boss.shootTimer = 40 - Math.min(20, score / 20);
      playShootSound(0.08);
    }
    if (boss.shootTimer <= 0) boss.shootTimer = 40;
  }
  
  // ======== اسپاون دشمنان ==========
  const enemyCap = Math.min(8, 3 + Math.floor(score / 80));
  if (enemies.length < enemyCap && Math.random() < 0.02) {
    spawnEnemy();
  }
  
  // ======== اسپاون باس ==========
  bossTimer++;
  if (bossTimer > 600 && !bossActive && score > 50) {
    spawnBoss();
    bossTimer = 0;
  }
  if (!bossActive && score > 50) bossTimer = Math.min(bossTimer, 600);
  
  // ======== بروزرسانی آیتمها ==========
  items.forEach((item, idx) => {
    item.life--;
    if (item.life <= 0) { items.splice(idx, 1); return; }
    drawItem(item);
    
    // جمعآوری آیتم
    if (distance(hero.x, hero.y, item.x, item.y) < hero.radius + item.radius + 5) {
      if (item.type === 'health') {
        hero.health = Math.min(hero.maxHealth, hero.health + 20);
        document.getElementById('healthDisplay').innerHTML = Math.floor(hero.health);
        createExplosion(item.x, item.y, 0.5, '#00ff88');
        playShootSound(0.05);
      } else {
        // قدرت: افزایش سرعت تیر
        hero.attackCooldown = Math.max(2, hero.attackCooldown - 1);
        createExplosion(item.x, item.y, 0.5, '#ffaa00');
        playShootSound(0.05);
      }
      items.splice(idx, 1);
    }
  });
  
  // ======== رسم همه چیز ========
  // آیتمها
  items.forEach(item => drawItem(item));
  
  // دشمنان
  enemies.forEach(e => drawEnemy(e));
  
  // باس
  if (bossActive && boss) drawBoss(boss);
  
  // گلولهها
  bullets.forEach(b => {
    ctx.shadowColor = b.color;
    ctx.shadowBlur = b.isRocket ? 40 : 20;
    ctx.beginPath();
    ctx.arc(b.x, b.y, b.radius, 0, Math.PI * 2);
    ctx.fillStyle = b.color;
    ctx.fill();
    ctx.shadowBlur = 0;
  });
  
  // قهرمان
  drawHero(hero.x, hero.y, hero.angle);
  
  // ======== افکت فلاش ==========
  if (screenFlash > 0) {
    ctx.fillStyle = `rgba(255,255,255,${screenFlash * 0.3})`;
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    screenFlash *= 0.95;
    if (screenFlash < 0.01) screenFlash = 0;
  }
  
  // ======== لرزش دوربین ==========
  if (shakeAmount > 0.1) {
    const shakeX = (Math.random() - 0.5) * shakeAmount;
    const shakeY = (Math.random() - 0.5) * shakeAmount;
    canvas.style.transform = `translate(${shakeX}px, ${shakeY}px)`;
    shakeAmount *= 0.9;
  } else {
    canvas.style.transform = 'translate(0, 0)';
    shakeAmount = 0;
  }
  
  // ======== نمایش کامبو ==========
  if (comboCount > 0) {
    ctx.fillStyle = `rgba(255,200,50,${0.3 + Math.sin(gameTime * 5) * 0.1})`;
    ctx.font = 'bold 24px Arial';
    ctx.textAlign = 'center';
    ctx.shadowColor = '#ffaa00';
    ctx.shadowBlur = 30;
    ctx.fillText(`🔥 ${comboCount}x COMBO!`, canvas.width/2, 60);
    ctx.shadowBlur = 0;
  }
  // کاهش کامبو با گذشت زمان
  if (gameTime % 60 < 0.5 && comboCount > 0) {
    comboCount--;
  }
  
  // ======== Game Over ==========
  if (hero.health <= 0) {
    ctx.fillStyle = 'rgba(0,0,0,0.8)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = '#ff2244';
    ctx.font = 'bold 60px Arial Black';
    ctx.textAlign = 'center';
    ctx.shadowColor = '#ff0000';
    ctx.shadowBlur = 50;
    ctx.fillText('💀 GAME OVER', canvas.width/2, canvas.height/2 - 30);
    ctx.shadowBlur = 0;
    ctx.fillStyle = 'rgba(255,255,255,0.3)';
    ctx.font = '20px monospace';
    ctx.fillText(`امتیاز: ${score}  |  کامبو: ${comboDisplay}`, canvas.width/2, canvas.height/2 + 50);
    ctx.fillStyle = 'rgba(255,255,255,0.15)';
    ctx.font = '16px monospace';
    ctx.fillText('برای شروع دوباره کلیک کن', canvas.width/2, canvas.height/2 + 95);
    return;
  }
  
  // ======== HUD ==========
  document.getElementById('scoreDisplay').innerHTML = score;
  document.getElementById('healthDisplay').innerHTML = Math.floor(hero.health);
  document.getElementById('comboDisplay').innerHTML = comboCount;
  
  requestAnimationFrame(animate);
}
